Set the channel event handler. After setting the channel event handler, you can listen for channel events and receive the statistics of the corresponding RtcChannel instance. Param handler The event handler.

Implementation

void setEventHandler(RtcChannelEventHandler handler) {
  _handler = handler;
  _subscription ??= _eventChannel.receiveBroadcastStream().listen((event) {
    final eventMap = Map<dynamic, dynamic>.from(event);
    final channelId = eventMap['channelId'];
    final methodName = eventMap['methodName'] as String;
    final data = List<dynamic>.from(eventMap['data']);
    _channels[channelId]?._handler?.process(methodName, data);
  });
}
